def display_mask(i):
    """Quick utility to display a model's prediction."""
    i = np.reshape(np.array(PIL.Image.open(i).resize((160,160))), (1,160,160,3))
    mask = np.argmax(model_obj.model.predict(i), axis = -1)
    mask = np.expand_dims(mask, axis = -1)
    img = tf.keras.preprocessing.image.array_to_img(mask[0])
    display(img)
